    40→/* ─── Analysis Error Boundary ────────────────────────────────── */
    41→interface ErrorBoundaryProps { children: ReactNode; onError: (msg: string) => void }
    42→interface ErrorBoundaryState { hasError: boolean; error: string | null }
    43→class AnalysisErrorBoundary extends Component<ErrorBoundaryProps, ErrorBoundaryState> {
    44→  constructor(props: ErrorBoundaryProps) { super(props); this.state = { hasError: false, error: null } }
    45→  static getDerivedStateFromError(err: unknown) {
    46→    const msg = err instanceof Error ? err.message : String(err)
    47→    return { hasError: true, error: msg }
    48→  }
    49→  componentDidCatch(err: unknown, info: ErrorInfo) {
    50→    console.error('[SpikeLab] AnalysisView render error:', err, info.componentStack)
    51→    this.props.onError(err instanceof Error ? err.message : String(err))
    52→  }
    53→  render() {
    54→    if (this.state.hasError) {
    55→      return (
    56→        <div className="bg-destructive/10 text-destructive rounded-lg p-6 text-center">
    57→          <AlertTriangle className="w-8 h-8 mx-auto mb-3" />
    58→          <p className="font-medium mb-1">Rendering Error</p>
    59→          <p className="text-sm text-destructive/80 mb-3">{this.state.error}</p>
    60→          <p className="text-xs text-muted-foreground">Please try again with a different video.</p>
    61→        </div>
    62→      )
    63→    }
    64→    return this.props.children
    65→  }
    66→}

   541→/* ─── Analysis View ──────────────────────────────────────────── */
   542→
   543→function AnalysisView({
   544→  analysis,
   545→  playerName,
   546→  onGeneratePlan,
   547→  isGenerating,
   548→  onReset,
   549→}: {
   550→  analysis: SpikeAnalysis
   551→  playerName: string
   552→  onGeneratePlan: () => void
   553→  isGenerating: boolean
   554→  onReset: () => void
   555→}) {
   556→  const { t } = useI18n()
   557→
   558→  const scoreValues = Object.values(analysis?.scores ?? {}).filter((v): v is number => typeof v === 'number')
   559→  const overallAvg = scoreValues.length > 0
   560→    ? Math.round(scoreValues.reduce((a, b) => a + b, 0) / scoreValues.length)
   561→    : 0
   562→
   563→  const phases = [
   564→    { key: 'approach' as const, label: t().analysis.phaseApproach, labelKey: 'phaseLabelApproach' as const, icon: Footprints, color: 'text-blue-500' },
   565→    { key: 'jump' as const, label: t().analysis.phaseJump, labelKey: 'phaseLabelJump' as const, icon: Activity, color: 'text-violet-500' },
   566→    { key: 'contact' as const, label: t().analysis.phaseContact, labelKey: 'phaseLabelContact' as const, icon: Flame, color: 'text-orange-500' },
   567→    { key: 'followThrough' as const, label: t().analysis.phaseFollowThrough, labelKey: 'phaseLabelFollowThrough' as const, icon: ShieldCheck, color: 'text-emerald-500' },
   568→  ]
   569→
   570→  const phaseLabelMap: Record<string, string> = {
   571→    approach: t().analysis.phaseLabelApproach,
   572→    jump: t().analysis.phaseLabelJump,
   573→    contact: t().analysis.phaseLabelContact,
   574→    followThrough: t().analysis.phaseLabelFollowThrough,
   575→  }
   576→
   577→  function getScoreLabel(score: number): string {
   578→    if (score >= 90) return t().scoreLabels.elite
   579→    if (score >= 76) return t().scoreLabels.excellent
   580→    if (score >= 60) return t().scoreLabels.decent
   581→    if (score >= 40) return t().scoreLabels.needsWork
   582→    return t().scoreLabels.critical
   583→  }
   584→
   585→  const checkpoints = t().checkpoints
   586→
   587→  return (
   588→    <div className="space-y-6">
   589→      {/* Overall Score */}
   590→      <Card className="overflow-hidden">
   591→        <div className="bg-gradient-to-r from-orange-500/10 via-amber-500/5 to-transparent p-6 sm:p-8">
   592→          <div className="flex flex-col sm:flex-row items-center gap-6">
   593→            <div className="relative w-28 h-28 sm:w-32 sm:h-32">
   594→              <svg viewBox="0 0 120 120" className="w-full h-full -rotate-90">
   595→                <circle cx="60" cy="60" r="52" fill="none" stroke="currentColor" className="text-muted/30" strokeWidth="8" />
   596→                <circle
   597→                  cx="60" cy="60" r="52" fill="none"
   598→                  stroke="currentColor" strokeWidth="8" strokeLinecap="round"
   599→                  strokeDasharray={`${(overallAvg / 100) * 327} 327`}
   600→                  className={getScoreColor(overallAvg)}
   601→                />
   602→              </svg>
   603→              <div className="absolute inset-0 flex flex-col items-center justify-center">
   604→                <span className="text-3xl font-bold">{overallAvg}</span>
   605→                <span className="text-xs text-muted-foreground">/ 100</span>
   606→              </div>
   607→            </div>
   608→            <div className="text-center sm:text-left flex-1">
   609→              <div className="flex items-center gap-2 mb-2">
   610→                <Badge variant="secondary">{analysis.estimatedLevel || 'intermediate'}</Badge>
   611→                <Badge variant="outline" className="text-xs font-normal">
   612→                  YOLOv8 Pose Tracking
   613→                </Badge>
   614→              </div>
   615→              <h2 className="text-xl sm:text-2xl font-bold mb-1">
   616→                {playerName ? `${playerName}'s` : t().analysis.yourSpikeAnalysis}
   617→              </h2>
   618→              <p className="text-muted-foreground text-sm leading-relaxed">
   619→                {analysis.coachNotes || ''}
   620→              </p>
   621→            </div>
   622→          </div>
   623→        </div>
   624→      </Card>
   625→
   626→      {/* Phase Scores */}
   627→      <div className="grid sm:grid-cols-2 gap-4">
   628→        {phases.map(({ key, label, icon: Icon, color }) => {
   629→          const phase = analysis.phaseAnalysis?.[key]
   630→          if (!phase) return null
   631→          const score = typeof phase.score === 'number' ? phase.score : 50
   632→          return (
   633→            <Card key={key} className="p-5">
   634→              <div className="flex items-center justify-between mb-3">
   635→                <div className="flex items-center gap-2">
   636→                  <Icon className={`w-5 h-5 ${color}`} />
   637→                  <h3 className="font-semibold">{label}</h3>
   638→                </div>
   639→                <span className={`text-2xl font-bold ${getScoreColor(score)}`}>{score}</span>
   640→              </div>
   641→              <Progress value={score} className="h-2 mb-3" />
   642→              <p className="text-sm text-muted-foreground">{phase.feedback || ''}</p>
   643→            </Card>
   644→          )
   645→        })}
   646→      </div>
   647→
   648→      {/* Detailed Checkpoint Scores */}
   649→      <Card>
   650→        <CardHeader>
   651→          <CardTitle className="text-lg">{t().analysis.allCheckpoints}</CardTitle>
   652→        </CardHeader>
   653→        <CardContent className="space-y-3">
   654→          {(['approach', 'jump', 'contact', 'followThrough'] as const).map(phase => {
   655→            const scoresObj = analysis.scores ?? {}
   656→            const phaseCheckpoints = Object.entries(scoresObj)
   657→              .filter(([k, v]) => typeof v === 'number' && CHECKPOINT_LABELS[k as keyof typeof CHECKPOINT_LABELS]?.phase === phase)
   658→            const phaseLabel = phaseLabelMap[phase] || phase
   659→            if (phaseCheckpoints.length === 0) return null
   660→            return (
   661→              <div key={phase}>
   662→                <h4 className="text-sm font-semibold text-muted-foreground uppercase tracking-wider mb-2">
   663→                  {phaseLabel}
   664→                </h4>
   665→                <div className="grid sm:grid-cols-2 gap-2">
   666→                  {phaseCheckpoints.map(([key, score]) => {
   667→                    const label = checkpoints[key as keyof typeof checkpoints] ?? key
   668→                    return (
   669→                      <div key={key} className="flex items-center justify-between bg-muted/30 rounded-lg px-3 py-2">
   670→                        <span className="text-sm">{label}</span>
   671→                        <div className="flex items-center gap-2">
   672→                          <span className={`text-sm font-semibold ${getScoreColor(score as number)}`}>{score}</span>
   673→                          <Badge variant="outline" className={`text-[10px] ${getScoreColor(score as number)}`}>
   674→                            {getScoreLabel(score as number)}
   675→                          </Badge>
   676→                        </div>
   677→                      </div>
   678→                    )
   679→                  })}
   680→                </div>
   681→              </div>
   682→            )
   683→          })}
   684→        </CardContent>
   685→      </Card>
   686→
   687→      {/* Strengths & Weaknesses */}
   688→      <div className="grid sm:grid-cols-2 gap-4">
   689→        <Card className="p-5">
   690→          <h3 className="font-semibold mb-3 flex items-center gap-2">
   691→            <CheckCircle2 className="w-5 h-5 text-emerald-500" /> {t().analysis.topStrengths}
   692→          </h3>
   693→          <ul className="space-y-2">
   694→            {(analysis.topStrengths ?? []).map((s, i) => (
   695→              <li key={i} className="text-sm text-muted-foreground flex gap-2">
   696→                <CheckCircle2 className="w-4 h-4 text-emerald-500 shrink-0 mt-0.5" />
   697→                <span>{s}</span>
   698→              </li>
   699→            ))}
   700→          </ul>
   701→        </Card>
   702→        <Card className="p-5">
   703→          <h3 className="font-semibold mb-3 flex items-center gap-2">
   704→            <AlertTriangle className="w-5 h-5 text-amber-500" /> {t().analysis.topWeaknesses}
   705→          </h3>
   706→          <ul className="space-y-2">
   707→            {(analysis.topWeaknesses ?? []).map((w, i) => (
   708→              <li key={i} className="text-sm text-muted-foreground flex gap-2">
   709→                <AlertTriangle className="w-4 h-4 text-amber-500 shrink-0 mt-0.5" />
   710→                <span>{w}</span>
   711→              </li>
   712→            ))}
   713→          </ul>
   714→        </Card>
   715→      </div>
   716→
   717→      {/* Actions */}
   718→      <div className="flex flex-col sm:flex-row gap-3">
   719→        <Button size="lg" className="flex-1 gap-2" onClick={onGeneratePlan} disabled={isGenerating}>
   720→          {isGenerating ? (
   721→            <>
   722→              <motion.div animate={{ rotate: 360 }} transition={{ duration: 1, repeat: Infinity, ease: 'linear' }}>
   723→                <RotateCcw className="w-4 h-4" />
   724→              </motion.div>
   725→              {t().analysis.generatingBtn}
   726→            </>
   727→          ) : (
   728→            <>{t().analysis.generateBtn} <ArrowRight className="w-4 h-4" /></>
   729→          )}
   730→        </Button>
   731→        <Button variant="outline" size="lg" onClick={onReset}>
   732→          {t().analysis.newAnalysisBtn}
   733→        </Button>
   734→      </div>
   735→    </div>
   736→  )
   737→}

   739→/* ─── Training Plan View ─────────────────────────────────────── */
   740→
   741→function TrainingPlanView({
   742→  plan,
   743→  onReset,
   744→}: {
   745→  plan: TrainingPlan
   746→  onReset: () => void
   747→}) {
   748→  const { t } = useI18n()
   749→  const [openVideo, setOpenVideo] = useState<string | null>(null)
   750→
   751→  function getYouTubeId(url: string): string | null {
   752→    const match = url.match(/(?:youtube\.com\/(?:watch\?v=|embed\/|v\/)|youtu\.be\/)([a-zA-Z0-9_-]{11})/)
   753→    return match ? match[1] : null
   754→  }
   755→
   756→  function toggleVideo(drillKey: string) {
   757→    setOpenVideo(prev => prev === drillKey ? null : drillKey)
   758→  }
   759→
   760→  return (
   761→    <div className="space-y-6">
   762→      <Card className="bg-gradient-to-r from-orange-500/10 via-amber-500/5 to-transparent p-6 sm:p-8">
   763→        <h2 className="text-xl sm:text-2xl font-bold mb-2">{t().training.title}</h2>
   764→        <p className="text-muted-foreground mb-4">{plan.summary}</p>
   765→        <div className="flex flex-wrap gap-2">
   766→          {plan.keyFocus.map(f => (
   767→            <Badge key={f} variant="secondary"><Target className="w-3 h-3 mr-1" />{f}</Badge>
   768→          ))}
   769→        </div>
   770→      </Card>
   771→
   772→      <AnimatePresence>
   773→        {plan.weeks.map((week) => (
   774→          <motion.div
   775→            key={week.week}
   776→            initial={{ opacity: 0, y: 20 }}
   777→            animate={{ opacity: 1, y: 0 }}
   778→            transition={{ delay: week.week * 0.1 }}
   779→          >
   780→            <Card className="overflow-hidden">
   781→              <CardHeader className="bg-muted/30">
   782→                <div className="flex items-center justify-between">
   783→                  <div>
   784→                    <Badge variant="outline" className="mb-1">{t().training.week} {week.week}</Badge>
   785→                    <CardTitle className="text-lg">{week.title}</CardTitle>
   786→                  </div>
   787→                </div>
   788→                <p className="text-sm text-muted-foreground">{week.focus}</p>
   789→              </CardHeader>
   790→              <CardContent className="p-4">
   791→                <div className="space-y-3">
   792→                  {week.days.map((day) => (
   793→                    <div key={day.day} className="border rounded-lg p-3">
   794→                      <div className="flex items-center justify-between mb-2">
   795→                        <h4 className="font-medium text-sm">{day.day}</h4>
   796→                        <Badge variant="secondary" className="text-xs">{day.phase}</Badge>
   797→                      </div>
   798→                      <div className="space-y-2">
   799→                        {day.drills.map((drill, i) => {
   800→                          const videoId = drill.videoUrl ? getYouTubeId(drill.videoUrl) : null
   801→                          const drillKey = `${week.week}-${day.day}-${i}`
   802→                          const isOpen = openVideo === drillKey
   803→                          const needsEquipment = drill.equipment && drill.equipment !== 'None'
   804→                          return (
   805→                            <div key={i} className="bg-muted/30 rounded-md p-2.5">
   806→                              <div className="flex items-start justify-between gap-2">
   807→                                <div className="flex-1 min-w-0">
   808→                                  <div className="flex items-center gap-2 flex-wrap">
   809→                                    <p className="text-sm font-medium">{drill.name}</p>
   810→                                    {needsEquipment && (
   811→                                      <span className="inline-flex items-center gap-1 text-[10px] font-medium text-amber-600 bg-amber-500/10 px-1.5 py-0.5 rounded-full shrink-0">
   812→                                        <Package className="w-3 h-3" />
   813→                                        {drill.equipment}
   814→                                      </span>
   815→                                    )}
   816→                                    {videoId && (
   817→                                      <button
   818→                                        onClick={() => toggleVideo(drillKey)}
   819→                                        className="inline-flex items-center gap-1 text-xs font-medium text-red-500 hover:text-red-400 transition-colors shrink-0"
   820→                                      >
   821→                                        <Youtube className="w-3.5 h-3.5" />
   822→                                        {isOpen ? t().training.hide : t().training.watch}
   823→                                        <ChevronDown className={`w-3 h-3 transition-transform ${isOpen ? 'rotate-180' : ''}`} />
   824→                                      </button>
   825→                                    )}
   826→                                  </div>
   827→                                  {drill.cue && (
   828→                                    <p className="text-xs text-muted-foreground mt-0.5 italic">
   829→                                      &quot;{drill.cue}&quot;
   830→                                    </p>
   831→                                  )}
   832→                                  {needsEquipment && drill.noEquipmentAlt && (
   833→                                    <div className="mt-1.5 flex items-start gap-1.5 bg-emerald-500/5 border border-emerald-500/20 rounded-md px-2 py-1.5">
   834→                                      <Lightbulb className="w-3 h-3 text-emerald-600 mt-0.5 shrink-0" />
   835→                                      <p className="text-[11px] text-emerald-700 dark:text-emerald-400 leading-relaxed">
   836→                                        <span className="font-semibold">{t().training.noEquipment}</span> {drill.noEquipmentAlt}
   837→                                      </p>
   838→                                    </div>
   839→                                  )}
   840→                                </div>
   841→                                <div className="text-right shrink-0">
   842→                                  <p className="text-xs font-medium">{drill.sets} × {drill.reps}</p>
   843→                                  {drill.duration && (
   844→                                    <p className="text-xs text-muted-foreground">{drill.duration}</p>
   845→                                  )}
   846→                                </div>
   847→                              </div>
   848→                              <AnimatePresence>
   849→                                {isOpen && videoId && (
   850→                                  <motion.div
   851→                                    initial={{ height: 0, opacity: 0 }}
   852→                                    animate={{ height: 'auto', opacity: 1 }}
   853→                                    exit={{ height: 0, opacity: 0 }}
   854→                                    transition={{ duration: 0.25, ease: 'easeInOut' }}
   855→                                    className="overflow-hidden"
   856→                                  >
   857→                                    <div className="mt-3 rounded-lg overflow-hidden border bg-black">
   858→                                      <iframe
   859→                                        className="w-full aspect-video"
   860→                                        src={`https://www.youtube-nocookie.com/embed/${videoId}?rel=0&modestbranding=1&start=${drill.videoStart || 0}`}
   861→                                        title={`${drill.name} Tutorial`}
   862→                                        allow="accelerometer; autoplay; clipboard-write; encrypted-media; gyroscope; picture-in-picture"
   863→                                        allowFullScreen
   864→                                      />
   865→                                    </div>
   866→                                  </motion.div>
   867→                                )}
   868→                              </AnimatePresence>
   869→                            </div>
   870→                          )
   871→                        })}
   872→                      </div>
   873→                    </div>
   874→                  ))}
   875→                </div>
   876→              </CardContent>
   877→            </Card>
   878→          </motion.div>
   879→        ))}
   880→      </AnimatePresence>
   881→
   882→      <div className="flex justify-center pt-4">
   883→        <Button variant="outline" size="lg" onClick={onReset} className="gap-2">
   884→          <RotateCcw className="w-4 h-4" /> {t().training.startOver}
   885→        </Button>
   886→      </div>
   887→    </div>
   888→  )
